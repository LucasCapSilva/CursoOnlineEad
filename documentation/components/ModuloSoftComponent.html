import { AlertModelService } from './../shared/alert-model.service';
import { Router } from '@angular/router';
import { Component, OnInit } from '@angular/core';
import { faForward } from '@fortawesome/free-solid-svg-icons';
import { faBackward } from '@fortawesome/free-solid-svg-icons';
import { environment } from './../../environments/environment.prod';
@Component({
  selector: 'app-modulo-soft',
  templateUrl: './modulo-soft.component.html',
  styleUrls: ['./modulo-soft.component.css']
})
export class ModuloSoftComponent implements OnInit {
  faForward = faForward;
  faBackward = faBackward;
  constructor(
    private router: Router,
    private alertService: AlertModelService) { }

  ngOnInit() {
    window.scroll(0, 0);

    const token = environment.token;

    if (token == '') {
      this.alertService.showAlertInfo('Antes de começar, faça o login!');
      this.router.navigate(['/login']);
    }
  }

  conteudo(tema: string) {
    environment.temaModulo = tema
    this.router.navigate(['/soft-conteudo']);
  }

  mostraTextOF(){
    if(document.getElementById('cardTextOF').style.display == 'block'){
      document.getElementById('cardTextOF').style.display = 'none'
    } else {
      document.getElementById('cardTextOF').style.display = 'block'
    }

  }

}
